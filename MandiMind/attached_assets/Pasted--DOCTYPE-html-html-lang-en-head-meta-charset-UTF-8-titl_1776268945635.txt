// ================= DROPDOWN LOGIC =================

function updateSelections(crop) {
  const varietyDropdown = document.getElementById("variety");
  const mandiDropdown = document.getElementById("mandi");

  varietyDropdown.innerHTML = "";
  mandiDropdown.innerHTML = "";

  if (!mandiData[crop]) return;

  mandiData[crop].varieties.forEach(v => {
    let opt = document.createElement("option");
    opt.value = v;
    opt.text = v;
    varietyDropdown.appendChild(opt);
  });

  mandiData[crop].mandis.forEach(m => {
    let opt = document.createElement("option");
    opt.value = m;
    opt.text = m;
    mandiDropdown.appendChild(opt);
  });
}

// ================= DECISION ENGINE =================

function getDecision(input) {
  let basePrice = 1000;

  if (input.crop === "Onion") basePrice = 1200;
  if (input.crop === "Tomato") basePrice = 800;
  if (input.crop === "Cotton") basePrice = 6000;

  if (input.quality === "High") basePrice += 300;
  if (input.quality === "Low") basePrice -= 250;

  if (input.variety === "Export Quality") basePrice += 500;
  if (input.variety === "Local") basePrice -= 100;

  if (input.urgency === "Need cash") basePrice -= 200;

  if (input.storage === "Yes" && input.urgency !== "Need cash") {
    return {
      advice: "Wait 5-7 days",
      expectedPrice: basePrice + 300
    };
  }

  return {
    advice: "Sell Now",
    expectedPrice: basePrice
  };
}

// ================= FINAL ACTION =================

function handleDecision() {
  const input = {
    crop: document.getElementById("crop").value,
    variety: document.getElementById("variety").value,
    mandi: document.getElementById("mandi").value,
    quality: document.getElementById("quality").value,
    storage: document.getElementById("storage").value,
    urgency: document.getElementById("urgency").value
  };

  const result = getDecision(input);

  document.getElementById("result").innerHTML = `
    <h3>Advice: ${result.advice}</h3>
    <p>Expected Price: ₹${result.expectedPrice}</p>
  `;
}
